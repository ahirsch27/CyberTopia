import pygame
import random
import time

# --- CONSTANTS ---
WIDTH, HEIGHT = 800, 500
FPS = 60
BLACK = (5, 5, 10)
CYBER_BLUE = (0, 255, 255)   
NEON_PINK = (255, 0, 255)    
WHITE = (255, 255, 255)
GOLD = (255, 215, 0)
GRID_GRAY = (20, 20, 35)

class Daemon:
    def __init__(self, name, hp, x, y, color, is_player=False):
        self.name = name
        self.level = 1
        self.exp = 0
        self.max_hp = hp
        self.hp = hp
        self.base_attack = 20
        self.x = x
        self.y = y
        self.color = color
        self.is_player = is_player

    def level_up(self):
        self.level += 1
        self.exp -= 100 # Reset XP but keep the carry-over
        self.max_hp += 15
        self.hp = self.max_hp # Full heal on level up
        self.base_attack += 5
        return f"SYSTEM UPGRADE: {self.name} IS NOW LVL {self.level}!"

    def draw(self, surface, offset=(0, 0)):
        draw_x, draw_y = self.x + offset[0], self.y + offset[1]
        font = pygame.font.SysFont("Courier", 20, bold=True)
        
        info_text = font.render(f"{self.name} [LVL {self.level}]", True, self.color)
        surface.blit(info_text, (draw_x, draw_y - 35))

        # Health Bar
        pygame.draw.rect(surface, (40, 40, 50), (draw_x, draw_y, 250, 20))
        hp_ratio = max(0, self.hp / self.max_hp)
        pygame.draw.rect(surface, self.color, (draw_x, draw_y, 250 * hp_ratio, 20))
        pygame.draw.rect(surface, WHITE, (draw_x, draw_y, 250, 20), 2)

def start_cyber_battle():
    pygame.init()
    screen = pygame.display.set_mode((WIDTH, HEIGHT))
    pygame.display.set_caption(">>> CYBERTOPIA: ENDLESS PROTOCOL <<<")
    clock = pygame.time.Clock()

    # 1. Initialize Player (Persistent)
    player = Daemon("GLITCH_CAT.exe", 100, 50, 350, CYBER_BLUE, True)
    
    # 2. Function to spawn a new enemy
    def spawn_enemy(player_level):
        names = ["CORP_SENTINEL", "ICE_BREAKER", "DATA_WRAITH", "NET_REAPER"]
        name = f"{random.choice(names)}.sys"
        # Enemy levels up slightly with the player
        e_hp = 70 + (player_level * 10)
        return Daemon(name, e_hp, 450, 100, NEON_PINK)

    enemy = spawn_enemy(player.level)
    
    log_msg = "ENEMIES DETECTED. [A] ATTACK | [S] REPAIR"
    running = True
    shake_intensity, shake_duration = 0, 0
    victory_pause = False
    pause_timer = 0

    while running:
        off_x, off_y = 0, 0
        if shake_duration > 0:
            off_x = random.randint(-shake_intensity, shake_intensity)
            off_y = random.randint(-shake_intensity, shake_intensity)
            shake_duration -= 1

        screen.fill(BLACK)
        # Background Grid
        for i in range(0, WIDTH, 40): pygame.draw.line(screen, GRID_GRAY, (i+off_x, 0), (i+off_x, HEIGHT))
        for i in range(0, HEIGHT, 40): pygame.draw.line(screen, GRID_GRAY, (0, i+off_y), (WIDTH, i+off_y))

        for event in pygame.event.get():
            if event.type == pygame.QUIT: running = False
            
            # Only allow input if we aren't in a victory pause or dead
            if not victory_pause and player.hp > 0:
                if event.type == pygame.KEYDOWN:
                    if event.key == pygame.K_a:
                        # Player Attack
                        dmg = random.randint(player.base_attack, player.base_attack + 10)
                        enemy.hp -= dmg
                        log_msg = f"DEALT {dmg} DMG!"
                        shake_intensity, shake_duration = 15, 8
                        
                        if enemy.hp <= 0:
                            # Start Victory Sequence
                            player.exp += 60
                            log_msg = "TARGET NEUTRALIZED. DOWNLOADING XP..."
                            victory_pause = True
                            pause_timer = pygame.time.get_ticks()
                        else:
                            # Quick Enemy Counter
                            e_dmg = random.randint(10 + player.level, 20 + player.level)
                            player.hp -= e_dmg
                            shake_intensity, shake_duration = 25, 12
                            if player.hp <= 0: log_msg = "CRITICAL FAILURE. CONNECTION LOST."

                    if event.key == pygame.K_s:
                        heal = 25 + (player.level * 2)
                        player.hp = min(player.max_hp, player.hp + heal)
                        log_msg = f"SYSTEM REPAIR: +{heal} HP"

        # 3. RESPAWN LOGIC
        if victory_pause:
            # Wait 2 seconds before spawning next enemy
            if pygame.time.get_ticks() - pause_timer > 2000:
                if player.exp >= 100:
                    log_msg = player.level_up()
                else:
                    log_msg = "NEW TARGET ACQUIRED. UPLINKING..."
                
                enemy = spawn_enemy(player.level)
                victory_pause = False

        # Render
        player.draw(screen, (off_x, off_y))
        if enemy.hp > 0: enemy.draw(screen, (off_x, off_y))

        # UI Text
        font = pygame.font.SysFont("Courier", 18)
        color = GOLD if "UPGRADE" in log_msg else WHITE
        msg_surf = font.render(log_msg, True, color)
        screen.blit(msg_surf, (WIDTH//2 - msg_surf.get_width()//2, HEIGHT - 50))

        pygame.display.flip()
        clock.tick(FPS)

    pygame.quit()

if __name__ == "__main__":
    start_cyber_battle()
